数据结构之线性表

线性表的链式存储实现

1.定义一个结构体

typedef struct lnode *list;
struct lnode{
    elementtype data;
    list next;
};
struct lnode l;
list ptrl;

2.求表长

int length(list ptrl){
list p = ptrl;//头结点
int j=0;
while(p)
{
    p=p->next;
    j++;
}
return j;
}

3.查找

（1）按序号查找：findkth

int findkth(list ptrl,int k){
    list p = ptrl;
    int j=1;
    while((p!=null)&&(j<k)){
        p=p->next;
        j++;
    }
    if(j==k) return p;//找到了，返回指针p
    else return null;//否则返回空
}

（2）按值查找：find

int find(list ptrl,elementtype x){
    list p = ptrl;
    while(p!=null&&p->data!=x)
        p=p->next;
    return p;
}

4.插入（在第i-1(l<=i<=n+1)个结点后插入一个值为x的新结点）

思路：1.先构造一个新节点，用s指向。 2.找到链表的第i-1个结点，用p指向 3.修改指针，插入结点

list insert(list ptrl,elementtype x,int i){
    list p,s;
    if(i==1){
        s=(list)malloc(sizeof(struct lnode));
        s->next = ptrl;
        s->data = x;
        return s;//返回新表头指针
    }
    p = findkth(ptrl,i-1);//找到第i-1个结点
    if(p==null){
        printf("未找到结点，参数i错误");
    return null;
    }
    else {
        s=(list)malloc(sizeof(struct lnode));
        s->next = p->next;
        p->next = s;
        s->data=x;
        return ptrl;
    }
}

5.删除（删除链表的第i(l<=i<=n)个位置上的结点)

思路：1.先找到链表第i-1个位置的结点，用p指向 2.找到要删除的结点，用s指向 3.修改指针，删除s结点 4.释放s结点的空间

list delete(list ptrl,int i){
    list p,s;
    if(i==1){
        s=ptrl;
        if(ptrl!=null) ptrl = ptrl->next;
        else return null;
        free(s);
        return ptrl;
    }
    p=findkth(ptrl,i-1);
    if(p==null){
        printf("第%d个结点不存在",i-1);  return null;
    }else if(p->next == null){ 
        printf("第%d个结点不存在",i);   return null;
    }else {
        s=p->next;
        p->next = s->next;
        free(s);
        return ptrl;
        }
}
